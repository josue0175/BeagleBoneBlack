  sync

#BOOT Method 2 copy all files to properly formatted card#

#1st create partitions/filesystems#
Partition 1:
Set the “boot” flag for the first partition.
type: FAT32
size: around 30MB
label: BOOT
flags: boot and lba

Partition2:
type: ext4
size: around 200MB, or remaining SD-card space
label: ROOT

#2nd copy bootloaders and rootfs to sd#
  ##cp tmp/deploy/images/beaglebone-yocto/{MLO,u-boot.img,zImage} /run/media/jgiles/boot/
  
  sudo tar x -C /run/media/jgiles/root -f core-image-minimal-beaglebone-yocto.tar.bz2 
  umount /run/media/jgiles/root 
  umount /run/media/jgiles/BOOT
